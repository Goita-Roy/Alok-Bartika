# ১. পছন্দের ৩টি কার্টুনের নাম দিয়ে একটি লিস্ট তৈরি করলাম
my_favorites = ["Tom and Jerry", "Ben 10", "Batman"]

print("--- 📜 আমার পছন্দের পুরো লিস্ট ---")
# ২. পুরো লিস্টটি একসাথে প্রিন্ট করলাম
print(my_favorites)

print("--------------------------------")

print("--- 🎯 লিস্টের একদম প্রথম কার্টুন ---")
# ৩. ইনডেক্স [0] ব্যবহার করে প্রথম উপাদানটি প্রিন্ট করলাম
print(my_favorites[0])
